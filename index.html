01
catalogo_integral
Estado de stock + precios mayoristas y minoristas por producto
VIEW ▾
Tablas base
productos, categorias, proveedores, inventario, precios_por_tipo
Función
Panel de catálogo con alerta automática de reposición de stock
Caso de uso
Dashboard de gestión de inventario y catálogo para administradores
CREATE OR REPLACE VIEW catalogo_integral AS
SELECT
    p.id_producto,
    p.nombre                           AS producto,
    c.nombre                           AS categoria,
    prov.nombre                        AS proveedor,
    p.precio_base                      AS precio_lista_minorista,
    COALESCE(pm.precio, p.precio_base)  AS precio_lista_mayorista,
    i.stock_actual,
    i.stock_minimo,
    CASE
        WHEN i.stock_actual = 0
            THEN 'Sin Stock'
        WHEN i.stock_actual <= i.stock_minimo
            THEN 'Alerta Reposición'
        ELSE 'Stock Óptimo'
    END AS estado_inventario,
    i.ubicacion
FROM productos p
INNER JOIN categorias c    ON p.id_categoria  = c.id_categoria
INNER JOIN proveedores prov ON p.id_proveedor  = prov.id_proveedor
INNER JOIN inventario i    ON p.id_producto   = i.id_producto
LEFT JOIN  precios_por_tipo pm
    ON pm.id_producto = p.id_producto
   AND pm.tipo_cliente = 'mayorista'
WHERE p.estado != 'inactivo';

analisis_cupones_clientes
Impacto financiero de cada cupón: redenciones, descuento e ingresos netos
VIEW ▾
Tablas base
cupones_descuento, uso_cupones, pedidos, clientes
Función
Análisis de rentabilidad por campaña promocional con agrupación por tipo de cliente
Caso de uso
Reporte de marketing para evaluar la efectividad de los cupones
CREATE OR REPLACE VIEW analisis_cupones_clientes AS
SELECT
    cup.codigo_promocional,
    cup.porcentaje_descuento,
    cup.estado                          AS estado_cupon,
    cl.tipo_cliente,
    COUNT(uso.id_uso)                   AS cantidad_redenciones,
    COALESCE(SUM(ped.subtotal), 0)    AS total_ventas_brutas,
    COALESCE(SUM(ped.descuento), 0)   AS capital_descontado,
    COALESCE(SUM(ped.total), 0)       AS ingresos_netos_generados

LEFT JOIN uso_cupones uso ON cup.id_cupon  = uso.id_cupon
LEFT JOIN pedidos ped     ON uso.id_pedido = ped.id_pedido
LEFT JOIN clientes cl     ON uso.id_cliente= cl.id_cliente
GROUP BY
    cup.id_cupon, cup.codigo_promocional,
    cup.porcentaje_descuento, cup.estado, cl.tipo_cliente
ORDER BY cantidad_redenciones DESC;
04
desempeno_comercial_productos
Ranking de productos por ingresos, unidades vendidas y rating de opiniones
VIEW ▾
Tablas base
productos, categorias, proveedores, detalle_pedido, pedidos, opiniones
Función
KPIs comerciales por producto: ingresos totales, presencia en pedidos y satisfacción promedio
Caso de uso
Informe ejecutivo de ventas y desempeño del catálogo
CREATE OR REPLACE VIEW desempeno_comercial_productos AS
SELECT
    p.id_producto,
    p.nombre                             AS producto,
    p.marca,
    c.nombre                             AS categoria,
    prov.nombre                          AS proveedor,
    COALESCE(SUM(dp.cantidad), 0)      AS total_unidades_vendidas,
    COALESCE(SUM(dp.subtotal), 0)      AS ingresos_totales,
    COUNT(DISTINCT ped.id_pedido)        AS presencia_en_pedidos,
    ROUND(AVG(o.calificacion), 2)        AS rating_satisfaccion
FROM productos p
INNER JOIN categorias c      ON c.id_categoria   = p.id_categoria
LEFT JOIN  proveedores prov  ON prov.id_proveedor = p.id_proveedor
LEFT JOIN  detalle_pedido dp ON dp.id_producto    = p.id_producto
LEFT JOIN  pedidos ped       ON ped.id_pedido     = dp.id_pedido
    AND ped.estado IN ('confirmado','en_preparacion','enviado','entregado')
LEFT JOIN  opiniones o       ON o.id_producto     = p.id_producto
    AND o.estado = 'visible'
GROUP BY   p.id_producto, p.nombre, p.marca, c.nombre, prov.nombre
ORDER BY   ingresos_totales DESC;
05
panel_financiero_ctacte
Estado crediticio de clientes mayoristas con saldo histórico y nivel de riesgo
VIEW ▾
Técnica avanzada
CTE + Window Function (ROW_NUMBER OVER PARTITION BY)
Función
Semáforo financiero: deuda actual, margen disponible y clasificación de riesgo crediticio
Caso de uso
Control de crédito para clientes mayoristas habilitados
CREATE OR REPLACE VIEW panel_financiero_ctacte AS
WITH SaldosHistoricos AS (
    SELECT
        id_cliente, saldo_posterior, fecha,
        ROW_NUMBER() OVER(
            PARTITION BY id_cliente
            ORDER BY fecha DESC, id_movimiento DESC
        ) AS fila
    FROM cuenta_corriente
)
SELECT
    cl.id_cliente,
    cl.razon_social,
    cl.cuit,
    CONCAT(u.nombre, ' ', u.apellido)    AS representante,
    cl.limite_credito                    AS credito_maximo_otorgado,
    COALESCE(sh.saldo_posterior, 0)     AS deuda_actual_exigible,
    (cl.limite_credito - COALESCE(sh.saldo_posterior, 0))
                                         AS margen_disponible,
    CASE
        WHEN COALESCE(sh.saldo_posterior,0) >= (cl.limite_credito * 0.9)
            THEN 'Riesgo Alto (Límite casi alcanzado)'
        WHEN COALESCE(sh.saldo_posterior,0) >= (cl.limite_credito * 0.5)
            THEN 'Riesgo Medio'
        ELSE 'Línea de Crédito Saludable'
    END AS estado_crediticio
FROM clientes cl
INNER JOIN usuarios u          ON u.id_usuario  = cl.id_cliente
LEFT JOIN  SaldosHistoricos sh ON sh.id_cliente = cl.id_cliente AND sh.fila = 1
WHERE cl.tipo_cliente = 'mayorista';

TRIGGERS

Procedimientos automáticos disparados por eventos DML. Garantizan integridad de datos sin depender de la lógica de la aplicación.

01
validacion_detalle_pedido
BEFORE INSERT — Valida stock e impide vender productos inactivos
BEFORE INSERT ▾
Tabla objetivo
detalle_pedido
Tipo
BEFORE INSERT · FOR EACH ROW
Protección
Rechaza filas con stock insuficiente o estado inactivo. Calcula subtotal automáticamente.
CREATE TRIGGER validacion_detalle_pedido
BEFORE INSERT ON detalle_pedido
FOR EACH ROW
BEGIN
    DECLARE v_estado_producto VARCHAR(20);
    DECLARE v_stock_disponible INT;

    SELECT estado        INTO v_estado_producto
    FROM productos       WHERE id_producto = NEW.id_producto;

    SELECT stock_actual   INTO v_stock_disponible
    FROM inventario      WHERE id_producto = NEW.id_producto;

    IF v_estado_producto = 'inactivo' THEN
        SIGNAL SQLSTATE '45000'
            SET MESSAGE_TEXT = 'No se puede vender un producto inactivo.';
    END IF;

    IF v_stock_disponible < NEW.cantidad THEN
        SIGNAL SQLSTATE '45000'
            SET MESSAGE_TEXT = 'Inventario insuficiente para la cantidad solicitada.';
    END IF;

    SET NEW.subtotal = NEW.cantidad * NEW.precio_unitario;
END$$
02
descontar_stock_venta_estricto
AFTER INSERT — Descuenta stock y registra movimiento automáticamente
AFTER INSERT ▾
Tabla objetivo
detalle_pedido
Tipo
AFTER INSERT · FOR EACH ROW
Automatización
Actualiza inventario y genera entrada en movimientos_stock con referencia al pedido
CREATE TRIGGER descontar_stock_venta_estricto
AFTER INSERT ON detalle_pedido
FOR EACH ROW
BEGIN
    DECLARE v_stock_anterior  INT;
    DECLARE v_stock_posterior INT;
    DECLARE v_id_cliente      INT;
    DECLARE v_numero_pedido   VARCHAR(30);

    SELECT stock_actual INTO v_stock_anterior
    FROM inventario WHERE id_producto = NEW.id_producto;

    SET v_stock_posterior = v_stock_anterior - NEW.cantidad;

    UPDATE inventario SET stock_actual = v_stock_posterior
    WHERE id_producto = NEW.id_producto;

    SELECT id_cliente, numero_pedido
    INTO  v_id_cliente, v_numero_pedido
    FROM  pedidos WHERE id_pedido = NEW.id_pedido;

    INSERT INTO movimientos_stock (
        id_producto, tipo_movimiento, cantidad,
        stock_anterior, stock_posterior, referencia, id_usuario
    ) VALUES (
        NEW.id_producto, 'salida_venta', NEW.cantidad,
        v_stock_anterior, v_stock_posterior,
        CONCAT('Salida automática - Pedido N° ', v_numero_pedido),
        v_id_cliente
    );
END$$
03
control_estado_por_stock
AFTER UPDATE — Cambia el estado del producto a "agotado" si el stock llega a 0
AFTER UPDATE ▾
Tabla objetivo
inventario
Tipo
AFTER UPDATE · FOR EACH ROW
Lógica
Bidireccional: marca como "agotado" si stock = 0, y restaura "activo" si se repone
CREATE TRIGGER control_estado_por_stock
AFTER UPDATE ON inventario
FOR EACH ROW
BEGIN
    DECLARE v_estado_actual VARCHAR(20);
    SELECT estado INTO v_estado_actual
    FROM productos WHERE id_producto = NEW.id_producto;

    -- Si queda sin stock y estaba activo â†’ agotado
    IF NEW.stock_actual <= 0 AND v_estado_actual = 'activo' THEN
        UPDATE productos SET estado = 'agotado'
        WHERE id_producto = NEW.id_producto;
    END IF;

    -- Si vuelve a tener stock y estaba agotado â†’ activo
    IF NEW.stock_actual > 0 AND v_estado_actual = 'agotado' THEN
        UPDATE productos SET estado = 'activo'
        WHERE id_producto = NEW.id_producto;
    END IF;
END$$
04
sincronizacion_logistica_pedidos
AFTER UPDATE — Sincroniza el estado del pedido cuando cambia el estado del envío
AFTER UPDATE ▾
Tabla objetivo
seguimiento_envios
Tipo
AFTER UPDATE · FOR EACH ROW
Integración
Propaga automáticamente los cambios logísticos al estado del pedido en la tabla pedidos
CREATE TRIGGER sincronizacion_logistica_pedidos
AFTER UPDATE ON seguimiento_envios
FOR EACH ROW
BEGIN
    -- Si el envío se marcó como entregado â†’ el pedido también
    IF  NEW.estado_envio = 'entregado'
    AND OLD.estado_envio != 'entregado' THEN
        UPDATE pedidos SET estado = 'entregado'
        WHERE id_pedido = NEW.id_pedido;
    END IF;

    -- Si el envío fue despachado â†’ el pedido pasa a "enviado"
    IF  NEW.estado_envio = 'despachado'
    AND OLD.estado_envio != 'despachado' THEN
        UPDATE pedidos SET estado = 'enviado'
        WHERE id_pedido = NEW.id_pedido;
    END IF;
END$$

01
registrar_entrada_mercaderia
Ingresa stock con transacción, valida proveedor y registra movimiento
TRANSACCIÓN ▾
Parámetros IN
p_id_producto, p_cantidad, p_id_proveedor, p_id_usuario, p_referencia_remito
Lógica
START TRANSACTION · UPDATE inventario · INSERT movimientos_stock · COMMIT / ROLLBACK
Validaciones
Cantidad > 0 · El producto debe pertenecer al proveedor indicado
CREATE PROCEDURE registrar_entrada_mercaderia(
    IN p_id_producto       INT,
    IN p_cantidad          INT,
    IN p_id_proveedor      INT,
    IN p_id_usuario        INT,
    IN p_referencia_remito TEXT
)
BEGIN
    DECLARE v_stock_ant  INT;
    DECLARE v_stock_post INT;
    DECLARE v_existe     INT;

    DECLARE EXIT HANDLER FOR SQLEXCEPTION
    BEGIN ROLLBACK; SIGNAL SQLSTATE '45000'
        SET MESSAGE_TEXT = 'Error. Transacción revertida.';
    END;

    IF p_cantidad <= 0 THEN
        SIGNAL SQLSTATE '45000'
            SET MESSAGE_TEXT = 'La cantidad debe ser positiva.';
    END IF;

    SELECT COUNT(*) INTO v_existe
    FROM productos
    WHERE id_producto = p_id_producto AND id_proveedor = p_id_proveedor;

    IF v_existe = 0 THEN
        SIGNAL SQLSTATE '45000'
            SET MESSAGE_TEXT = 'El producto no pertenece al proveedor indicado.';
    END IF;

    START TRANSACTION;
    SELECT stock_actual INTO v_stock_ant
    FROM  inventario WHERE id_producto = p_id_producto FOR UPDATE;

    SET v_stock_post = v_stock_ant + p_cantidad;

    UPDATE inventario SET stock_actual = v_stock_post
    WHERE id_producto = p_id_producto;

    INSERT INTO movimientos_stock
        (id_producto, tipo_movimiento, cantidad,
         stock_anterior, stock_posterior, referencia, id_usuario)
    VALUES
        (p_id_producto, 'entrada', p_cantidad,
         v_stock_ant, v_stock_post, p_referencia_remito, p_id_usuario);

    COMMIT;
END$$
02
aplicar_cupon_promocional
Valida y aplica un cupón a un pedido, controlando uso previo del cliente
TRANSACCIÓN ▾
Parámetros IN
p_id_pedido, p_codigo, p_id_cliente
Lógica
Verifica vigencia del cupón · Chequea uso previo del cliente · Calcula y aplica descuento
Cálculo
descuento = ROUND(subtotal * porcentaje / 100, 2) · total = subtotal - descuento
CREATE PROCEDURE aplicar_cupon_promocional(
    IN p_id_pedido INT,
    IN p_codigo    VARCHAR(50),
    IN p_id_cliente INT
)
BEGIN
    DECLARE v_id_cupon    INT;
    DECLARE v_porcentaje  DECIMAL(5,2);
    DECLARE v_subtotal    DECIMAL(12,2);
    DECLARE v_descuento   DECIMAL(12,2);
    DECLARE v_usos_previos INT;

    SELECT id_cupon, porcentaje_descuento
    INTO  v_id_cupon, v_porcentaje
    FROM  cupones_descuento
    WHERE codigo_promocional = p_codigo
      AND estado = 'activo'
      AND fecha_vencimiento >= CURRENT_DATE;

    IF v_id_cupon IS NULL THEN
        SIGNAL SQLSTATE '45000'
            SET MESSAGE_TEXT = 'Cupón inválido, inactivo o vencido.';
    END IF;

    SELECT COUNT(*) INTO v_usos_previos
    FROM uso_cupones
    WHERE id_cupon = v_id_cupon AND id_cliente = p_id_cliente;

    IF v_usos_previos > 0 THEN
        SIGNAL SQLSTATE '45000'
            SET MESSAGE_TEXT = 'El cliente ya redimió este cupón.';
    END IF;

    START TRANSACTION;
    SELECT subtotal INTO v_subtotal
    FROM  pedidos WHERE id_pedido = p_id_pedido FOR UPDATE;

    SET v_descuento = ROUND((v_subtotal * v_porcentaje) / 100, 2);

    UPDATE pedidos SET
        descuento = v_descuento,
        total     = subtotal - v_descuento
    WHERE id_pedido = p_id_pedido;

    INSERT INTO uso_cupones (id_cupon, id_pedido, id_cliente)
    VALUES (v_id_cupon, p_id_pedido, p_id_cliente);
    COMMIT;
END$$
03
generar_despacho_logistico
Crea el seguimiento de envío solo si el pago está aprobado
TRANSACCIÓN ▾
Parámetros IN
p_id_pedido, p_id_empresa
Lógica
Verifica estado del pedido y aprobación del pago · Genera código tracking único · Crea registro en seguimiento_envios
Tracking
CONCAT('TRACK-', empresa, '-', año, '-', LPAD(id_pedido, 6, '0'))
CREATE PROCEDURE generar_despacho_logistico(
    IN p_id_pedido  INT,
    IN p_id_empresa INT
)
BEGIN
    DECLARE v_estado_pedido  VARCHAR(20);
    DECLARE v_estado_pago    VARCHAR(20);
    DECLARE v_codigo_tracking VARCHAR(100);

    SELECT estado INTO v_estado_pedido
    FROM pedidos WHERE id_pedido = p_id_pedido;

    SELECT estado INTO v_estado_pago
    FROM pagos WHERE id_pedido = p_id_pedido LIMIT 1;

    IF v_estado_pedido IN ('cancelado', 'entregado') THEN
        SIGNAL SQLSTATE '45000'
            SET MESSAGE_TEXT = 'Estado final: no admite despacho.';
    END IF;

    IF v_estado_pago != 'aprobado' THEN
        SIGNAL SQLSTATE '45000'
            SET MESSAGE_TEXT = 'Pago no aprobado. Logística bloqueada.';
    END IF;

    START TRANSACTION;
    SET v_codigo_tracking = CONCAT(
        'TRACK-', p_id_empresa, '-',
        YEAR(CURRENT_DATE), '-',
        LPAD(p_id_pedido, 6, '0')
    );

    INSERT INTO seguimiento_envios
        (id_pedido, id_empresa, codigo_tracking, estado_envio)
    VALUES
        (p_id_pedido, p_id_empresa, v_codigo_tracking, 'en_preparacion');

    UPDATE pedidos SET estado = 'en_preparacion'
    WHERE id_pedido = p_id_pedido;
    COMMIT;
END$$
04
amortizar_deuda_ctacte
Registra un pago parcial en la cuenta corriente de un cliente mayorista
TRANSACCIÓN ▾
Parámetros IN
p_id_cliente, p_monto_abonado, p_referencia_bancaria
Restricciones
Solo mayoristas · Monto > 0 · No se aceptan pagos superiores a la deuda actual
Lógica
Obtiene saldo más reciente · Calcula saldo posterior · Inserta movimiento tipo "pago"
CREATE PROCEDURE amortizar_deuda_ctacte(
    IN p_id_cliente         INT,
    IN p_monto_abonado      DECIMAL(12,2),
    IN p_referencia_bancaria TEXT
)
BEGIN
    DECLARE v_tipo_cliente    VARCHAR(20);
    DECLARE v_saldo_anterior  DECIMAL(12,2);
    DECLARE v_saldo_posterior DECIMAL(12,2);

    SELECT tipo_cliente INTO v_tipo_cliente
    FROM clientes WHERE id_cliente = p_id_cliente;

    IF v_tipo_cliente != 'mayorista' THEN
        SIGNAL SQLSTATE '45000'
            SET MESSAGE_TEXT = 'Operación exclusiva para clientes Mayoristas.';
    END IF;

    START TRANSACTION;
    SELECT saldo_posterior INTO v_saldo_anterior
    FROM  cuenta_corriente
    WHERE id_cliente = p_id_cliente
    ORDER BY fecha DESC LIMIT 1 FOR UPDATE;

    SET v_saldo_posterior =
        COALESCE(v_saldo_anterior, 0) - p_monto_abonado;

    IF v_saldo_posterior < 0 THEN
        SIGNAL SQLSTATE '45000'
            SET MESSAGE_TEXT = 'El pago supera la deuda actual.';
    END IF;

    INSERT INTO cuenta_corriente
        (id_cliente, tipo_movimiento, monto,
         saldo_anterior, saldo_posterior, descripcion)
    VALUES
        (p_id_cliente, 'pago', p_monto_abonado,
         v_saldo_anterior, v_saldo_posterior, p_referencia_bancaria);
    COMMIT;
END$$
05
convertir_carrito_a_pedido
Convierte un carrito activo en pedido con pago, en una sola transacción atómica
TRANSACCIÓN ▾
Parámetros IN
p_id_usuario, p_id_metodo_pago, p_metodo_envio, p_direccion
Flujo atómico
Bloquea carrito â†’ calcula subtotal â†’ genera pedido â†’ copia ítems â†’ crea pago â†’ cierra carrito
Número de pedido
NS-{año}-{rand 99999} — único por transacción
CREATE PROCEDURE convertir_carrito_a_pedido(
    IN p_id_usuario     INT,
    IN p_id_metodo_pago INT,
    IN p_metodo_envio   VARCHAR(30),
    IN p_direccion      TEXT
)
BEGIN
    DECLARE v_id_carrito     INT;
    DECLARE v_id_pedido      INT;
    DECLARE v_subtotal_global DECIMAL(12,2);
    DECLARE v_nro_pedido     VARCHAR(30);

    START TRANSACTION;

    SELECT id_carrito INTO v_id_carrito
    FROM  carritos
    WHERE id_usuario = p_id_usuario AND estado = 'activo'
    ORDER BY fecha_creacion DESC LIMIT 1 FOR UPDATE;

    SELECT COALESCE(SUM(cantidad * precio_unitario), 0)
    INTO   v_subtotal_global
    FROM   items_carrito
    WHERE  id_carrito = v_id_carrito;

    SET v_nro_pedido = CONCAT('NS-',
        DATE_FORMAT(CURRENT_DATE, '%Y'), '-',
        FLOOR(RAND() * 99999));

    INSERT INTO pedidos
        (id_cliente, numero_pedido, estado, subtotal, descuento, total,
         metodo_envio, direccion_envio)
    VALUES
        (p_id_usuario, v_nro_pedido, 'pendiente',
         v_subtotal_global, 0, v_subtotal_global,
         p_metodo_envio, p_direccion);

    SET v_id_pedido = LAST_INSERT_ID();

    -- Copia masiva de ítems del carrito al detalle del pedido
    INSERT INTO detalle_pedido
        (id_pedido, id_producto, cantidad, precio_unitario, subtotal, descuento)
    SELECT v_id_pedido, id_producto, cantidad,
           precio_unitario, (cantidad * precio_unitario), 0
    FROM  items_carrito WHERE id_carrito = v_id_carrito;

    INSERT INTO pagos (id_pedido, id_metodo, monto, estado)
    VALUES (v_id_pedido, p_id_metodo_pago, v_subtotal_global, 'pendiente');

    UPDATE carritos SET estado = 'convertido'
    WHERE id_carrito = v_id_carrito;

    COMMIT;
END$$
⬡ Fundamentación Teórica

NORMALIZACIÓN

La base de datos NexoSport está diseñada cumpliendo las tres primeras formas normales (1FN, 2FN, 3FN), garantizando consistencia y eliminación de redundancias.

1FN
Primera Forma Normal
Todos los atributos contienen valores atómicos e indivisibles. No existen grupos repetitivos ni atributos multivaluados.
Cada columna almacena un único valor Todas las tablas tienen clave primaria definida No hay listas ni conjuntos dentro de campos El campo datos_extra JSON en pagos es atómico para el motor
2FN
Segunda Forma Normal
Cumple 1FN y todo atributo no clave depende completamente de la clave primaria entera. Sin dependencias parciales.
Tablas de herencia (clientes, vendedores, admins) usan PK = FK a usuarios precios_por_tipo: clave compuesta (producto + tipo_cliente + cantidad) detalle_pedido: cada campo depende del ítem completo, no solo del pedido items_carrito: UNIQUE (id_carrito, id_producto) previene duplicados
3FN
Tercera Forma Normal
Cumple 2FN y no existen dependencias transitivas entre atributos no clave. Cada dato depende solo de su PK.
Proveedores separados de productos: no se repite nombre/CUIT en cada producto Categorías en tabla propia: no hay datos de categoría en productos Empresas de transporte separadas de seguimiento_envios Métodos de pago normalizados en metodos_pago, referenciados por FK
Conceptos clave para la evaluación
Clave Primaria (PK)
Identifica unívocamente cada fila. Nunca nula. Todas las tablas la tienen declarada con AUTO_INCREMENT o como PK natural.
Clave Foránea (FK)
Garantiza integridad referencial. Usamos ON DELETE CASCADE en relaciones donde la entidad hija no tiene sentido sin la padre.
Índices
Se crearon índices explícitos en columnas de búsqueda frecuente: email, estado, id_categoria, id_cliente, fecha, código de tracking.
Restricciones CHECK
Aplicadas en campos de dominio cerrado: estado, tipo_cliente, tipo_movimiento, calificacion (1–5), porcentaje_descuento (>0 y ≤100).

SELECT
SELECT — Consultas de datos
Tres consultas con JOINs múltiples, GROUP BY y filtros de negocio
DML▾ 
-- A: Estado de envíos activos en tiempo real
SELECT
    ped.numero_pedido,
    CONCAT(u.nombre, ' ', u.apellido) AS cliente,
    emp.nombre                          AS empresa_transporte,
    seg.codigo_tracking,
    seg.estado_envio,
    ped.metodo_envio
FROM pedidos ped
INNER JOIN clientes cl               ON ped.id_cliente = cl.id_cliente
INNER JOIN usuarios u                ON cl.id_cliente  = u.id_usuario
INNER JOIN seguimiento_envios seg    ON ped.id_pedido  = seg.id_pedido
INNER JOIN empresas_transporte emp   ON seg.id_empresa = emp.id_empresa
WHERE ped.estado NOT IN ('entregado', 'cancelado')
ORDER BY ped.fecha DESC;

-- B: Impacto de cupones de descuento en ventas
SELECT
    cup.codigo_promocional,
    cup.porcentaje_descuento,
    COUNT(uso.id_uso)       AS veces_utilizado,
    SUM(ped.descuento)      AS total_dinero_descontado,
    SUM(ped.total)          AS total_ingresos_generados
FROM cupones_descuento cup
LEFT JOIN uso_cupones uso ON cup.id_cupon  = uso.id_cupon
LEFT JOIN pedidos ped     ON uso.id_pedido = ped.id_pedido
GROUP BY  cup.id_cupon, cup.codigo_promocional, cup.porcentaje_descuento
ORDER BY  veces_utilizado DESC;

-- C: Productos activos con stock por debajo del mínimo
SELECT
    p.nombre    AS producto,
    c.nombre    AS categoria,
    prov.nombre AS proveedor,
    i.stock_actual,
    p.precio_base
FROM productos p
INNER JOIN categorias c      ON p.id_categoria  = c.id_categoria
LEFT JOIN  proveedores prov  ON p.id_proveedor   = prov.id_proveedor
INNER JOIN inventario i      ON p.id_producto    = i.id_producto
WHERE p.estado = 'activo'
  AND i.stock_actual < i.stock_minimo;

Limpieza de carritos abandonados y cupones vencidos, con Safe Updates
